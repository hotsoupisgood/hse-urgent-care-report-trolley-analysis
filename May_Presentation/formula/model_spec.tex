\textbf{Likelihood:}
\[
y_{i,t} \sim \mathcal{N}\!\left(\mu_{i,t} + \epsilon_{i,t},\; \tau_i\right), \quad t > 1
\]
\[
\textit{Where y is the raw weekly rate, i is the health region, and t is week}
\]  
\bigskip
\textbf{Mean structure:}
\begin{align*}
\mu_{i,t} =\;& \alpha_i 
   && \leftarrow \qquad \textbf{Baseline} \\
&+ \beta_i \cos\!\left(\tfrac{2\pi t}{52}\right) + \gamma_i \sin\!\left(\tfrac{2\pi t}{52}\right)
   && \leftarrow \qquad \textbf{Annual cycle} \\
&+ \begin{aligned}[t]
     & \delta_{i,t}\,\mathbf{1}(t = \text{New Year week}-2) + \delta_{i,t}\,\mathbf{1}(t = \text{New Year week}-1) \\
     & \phantom{\delta_{i,t}\,\mathbf{1}(t = \text{New Year week}-2)\ } + \delta_{i,t}\,\mathbf{1}(t = \text{New Year week}) \\
     & \phantom{\delta_{i,t}\,\mathbf{1}(t = \text{New Year week}-2)\ } + \delta_{i,t}\,\mathbf{1}(t = \text{New Year week}+1)
   \end{aligned}
   && \leftarrow \qquad \textbf{New Year effect} \\
&+ \begin{aligned}[t]
     & \kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Aug 8 2024 week}) + \kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Aug 15 2024 week}) \\
     & \phantom{\kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Aug 8 2024 week})\ } + \kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Aug 22 2024 week}) \\
     & \phantom{\kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Aug 8 2024 week})\ } + \kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Aug 29 2024 week}) \\
     & \phantom{\kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Aug 8 2024 week})\ } + \kappa_{i=\text{MW},\,t}\,\mathbf{1}(t = \text{Sep 5 2024 week})
   \end{aligned}
   && \leftarrow \qquad \textbf{Mid West reset}
\end{align*}
\bigskip
\textbf{AR(1) Component:}
\begin{align*}
\epsilon_{i,t} = \phi\,(y_{i,t-1} - \mu_{i,t-1})
\end{align*}
\bigskip
\textbf{Non-informative priors:}
\[
\alpha_i,\; \beta_i,\; \gamma_i,\; \delta_{i,t},\; \kappa_{i=\text{MW},\,t} \sim \mathcal{N}(0,\; 0.001)
\]
\[
\tau_i \sim \Gamma(0.001,\; 0.001) \qquad \phi \sim \mathrm{Uniform}(-1,\; 1)
\]
{\footnotesize\itshape Normal priors $\mathcal{N}(\mu,\tau)$ use precision in the 2nd argument.}\\
