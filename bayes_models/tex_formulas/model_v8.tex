\section*{Model V8 --- Hospital-Level Hierarchical}

\subsection*{Likelihood}

\begin{align}
  y_{j,1} &\sim \mathrm{NegBin}(\mu_{j,1},\ r_j) \\[6pt]
  y_{j,t} &\sim \mathrm{NegBin}(\mu_{j,t},\ r_j), \quad t > 1
\end{align}

where $\mathrm{Var}(y_{j,t}) = \mu_{j,t} + \mu_{j,t}^2 / r_j$.

\subsection*{$\mu$}

\begin{align}
  \log \mu_{j,1} &= \alpha_j
    + \beta_j^{(s)} \cos\!\left(\tfrac{2\pi t}{52}\right)
    + \gamma_j^{(s)} \sin\!\left(\tfrac{2\pi t}{52}\right)
    + \eta_{\text{pre}} \cdot \mathrm{ny}_{\text{pre},t}
    + \eta_{\text{mid}} \cdot \mathrm{ny}_{\text{mid},t}
    + \eta_{\text{post}} \cdot \mathrm{ny}_{\text{post},t} \\[6pt]
  \log \mu_{j,t} &= \underbrace{\alpha_j
    + \beta_j^{(s)} \cos\!\left(\tfrac{2\pi t}{52}\right)
    + \gamma_j^{(s)} \sin\!\left(\tfrac{2\pi t}{52}\right)
    + \eta_{\text{pre}} \cdot \mathrm{ny}_{\text{pre},t}
    + \eta_{\text{mid}} \cdot \mathrm{ny}_{\text{mid},t}
    + \eta_{\text{post}} \cdot \mathrm{ny}_{\text{post},t}}_{\text{base}_{j,t}} \nonumber \\
    &\quad + \psi \left(\log(y_{j,t-1} + 1) - \log \mu_{j,t-1}\right), \quad t > 1
\end{align}

\subsection*{Hierarchical Intercepts}

\begin{align}
  \mu_{\text{region},k} &\sim \mathcal{N}\!\left(\mu_{\text{global}} + \beta_{\text{pop}} \cdot \log\!\left(\tfrac{\mathrm{pop}_k}{10000}\right),\ \sigma_{\text{region}}^2\right) \\[4pt]
  \alpha_j &\sim \mathcal{N}\!\left(\mu_{\text{region},r(j)} + \delta_{\text{tier}} \cdot \mathbf{1}(\text{tier}_j = 4),\ \sigma_\alpha^2\right)
\end{align}

where $r(j)$ maps hospital $j$ to its HSE region.

\subsection*{Hierarchical Seasonal Priors}

\begin{align}
  \mu_{\beta,k}^{(s)},\ \mu_{\gamma,k}^{(s)} &\sim \mathcal{N}(0,\ 4) \\[4pt]
  \beta_j^{(s)} &\sim \mathcal{N}\!\left(\mu_{\beta,r(j)}^{(s)},\ \sigma_\beta^2\right) \\
  \gamma_j^{(s)} &\sim \mathcal{N}\!\left(\mu_{\gamma,r(j)}^{(s)},\ \sigma_\gamma^2\right)
\end{align}

\subsection*{Hyperpriors}

\begin{align}
  \mu_{\text{global}} &\sim \mathcal{N}(3,\ 9) \\
  \beta_{\text{pop}} &\sim \mathcal{N}(0,\ 4) \\
  \delta_{\text{tier}} &\sim \mathcal{N}(0,\ 4) \\
  \sigma_{\text{region}},\ \sigma_\alpha &\sim \mathrm{Uniform}(0,\ 5) \\
  \sigma_\beta,\ \sigma_\gamma &\sim \mathrm{Uniform}(0,\ 3)
\end{align}

\subsection*{Priors (global)}

\begin{align}
  \psi &\sim \mathrm{Uniform}(-1,\ 1) \\
  \eta_{\text{pre}},\ \eta_{\text{mid}},\ \eta_{\text{post}} &\sim \mathcal{N}(0,\ 4) \\
  r_j &\sim \Gamma(2,\ 0.5)
\end{align}
