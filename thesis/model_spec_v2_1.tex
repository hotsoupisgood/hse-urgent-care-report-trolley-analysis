  y_{i,1}   &\sim \mathcal{N}\!\left(\mu_{i,1},\ \text{precision}_i^{-1}\right) \\[6pt]
  y_{i,t}   &\sim \mathcal{N}\!\left(\mu_{i,t} + \text{AR}(1)\,(y_{i,t-1} - \mu_{i,t-1}),\ \text{precision}_i^{-1}\right), \quad t \geq 2 \\[12pt]
  \mu_{i,t} &= \text{baseline}_i \\
            &\quad + \text{annual cycle cos}_i \cdot \cos\!\left(\tfrac{2\pi t}{52}\right)
                  + \text{annual cycle sin}_i \cdot \sin\!\left(\tfrac{2\pi t}{52}\right) \\
            &\quad + \text{New Year pre}_i\,\mathbf{1}(t \bmod 52 = 0)
                  + \text{New Year mid}_i\,\mathbf{1}(t \bmod 52 = 1)
                  + \text{New Year post}_i\,\mathbf{1}(t \bmod 52 = 2) \\
            &\quad + \bigl(\text{MW reset pre}\,\mathbf{1}(t = 86)
                        + \text{MW reset mid}\,\mathbf{1}(t = 87)
                        + \text{MW reset post}\,\mathbf{1}(t = 88)\bigr)\cdot\mathrm{mw}_i \\[12pt]
  \text{baseline}_i,\ \text{annual cycle cos}_i,\ \text{annual cycle sin}_i
            &\sim \mathcal{N}(0,\ 1000) \\
  \text{New Year pre}_i,\ \text{New Year mid}_i,\ \text{New Year post}_i
            &\sim \mathcal{N}(0,\ 1000) \\
  \text{precision}_i
            &\sim \mathrm{Gamma}(0.001,\ 0.001) \\[6pt]
  \text{AR}(1)
            &\sim \mathrm{Uniform}(-1,\ 1) \\
  \text{MW reset pre},\ \text{MW reset mid},\ \text{MW reset post}
            &\sim \mathcal{N}(0,\ 1000)
\end{align}
