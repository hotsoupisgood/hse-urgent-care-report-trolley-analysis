\section*{Model V1 --- AR(1) with Seasonality}

\subsection*{Likelihood}

\begin{align}
  y_{i,1} &\sim \mathcal{N}(\mu_{i,1},\ \tau_i^{-1}) \\[6pt]
  y_{i,t} &\sim \mathcal{N}\left(\mu_{i,t} + \phi\,(y_{i,t-1} - \mu_{i,t-1}),\ \tau_i^{-1}\right), \quad t > 1
\end{align}

\subsection*{Mean Function}

\begin{equation}
  \mu_{i,t} = \alpha_i
    + \beta_i \cos\left(\tfrac{2\pi t}{52}\right)
    + \gamma_i \sin\left(\tfrac{2\pi t}{52}\right)
\end{equation}

\subsection*{Priors}

\begin{align}
  \alpha_i,\ \beta_i,\ \gamma_i &\sim \mathcal{N}(0,\ 0.001) \\
  \tau_i &\sim \Gamma(0.001,\ 0.001) \\
  \phi &\sim \mathrm{Uniform}(-1,\ 1)
\end{align}
