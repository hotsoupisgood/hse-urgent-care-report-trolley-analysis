\section*{Model V2c --- 6-Month Cycle + Event Effects}

\subsection*{Extends V2}

Replaces the annual cycle ($t/52$) with a 6-month cycle ($t/26$). All event effects unchanged.

\subsection*{Likelihood}

\begin{align}
  y_{i,1} &\sim \mathcal{N}(\mu_{i,1},\ \tau_i^{-1}) \\[6pt]
  y_{i,t} &\sim \mathcal{N}\left(\mu_{i,t} + \phi\,(y_{i,t-1} - \mu_{i,t-1}),\ \tau_i^{-1}\right), \quad t > 1
\end{align}

\subsection*{$\mu$}

\begin{multline}
  \mu_{i,t} = \alpha_i
    + \beta_i \cos\left(\tfrac{2\pi t}{26}\right)
    + \gamma_i \sin\left(\tfrac{2\pi t}{26}\right) \\
    + \delta_{\text{pre}} \cdot \mathbf{1}(t \bmod 52 = 0)
    + \delta_{\text{mid}} \cdot \mathbf{1}(t \bmod 52 = 1)
    + \delta_{\text{post}} \cdot \mathbf{1}(t \bmod 52 = 2) \\
    + \sigma_{\text{pre}} \cdot \mathbf{1}(t = 86) \cdot \mathrm{mw}_i
    + \sigma_{\text{mid}} \cdot \mathbf{1}(t = 87) \cdot \mathrm{mw}_i
    + \sigma_{\text{post}} \cdot \mathbf{1}(t = 88) \cdot \mathrm{mw}_i
\end{multline}

\subsection*{Priors (region-level)}

\begin{align}
  \alpha_i,\ \beta_i,\ \gamma_i &\sim \mathcal{N}(0,\ 0.001) \\
  \tau_i &\sim \Gamma(0.001,\ 0.001)
\end{align}

\subsection*{Priors (global)}

\begin{align}
  \phi &\sim \mathrm{Uniform}(-1,\ 1) \\
  \delta_{\text{pre}},\ \delta_{\text{mid}},\ \delta_{\text{post}} &\sim \mathcal{N}(0,\ 0.001) \\
  \sigma_{\text{pre}},\ \sigma_{\text{mid}},\ \sigma_{\text{post}} &\sim \mathcal{N}(0,\ 0.001)
\end{align}
